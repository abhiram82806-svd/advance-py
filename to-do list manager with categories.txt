to-do list manager with categories
add tasks , set deadlines , assign priority , and filter by project or tag

from datetime import datetime

class Task:
    def __init__(self, title, deadline, priority, project):
        self.title = title
        self.deadline = deadline
        self.priority = priority
        self.project = project

    def display(self):
        print(f"Task: {self.title}")
        print(f"Deadline: {self.deadline}")
        print(f"Priority: {self.priority}")
        print(f"Project: {self.project}")
        print("-" * 30)


class TodoManager:
    def __init__(self):
        self.tasks = []

    def add_task(self, title, deadline, priority, project):
        task = Task(title, deadline, priority, project)
        self.tasks.append(task)
        print("Task added successfully!")

    def show_all_tasks(self):
        if not self.tasks:
            print("No tasks available.")
        for task in self.tasks:
            task.display()

    def filter_by_project(self, project_name):
        found = False
        for task in self.tasks:
            if task.project == project_name:
                task.display()
                found = True
        if not found:
            print("No tasks found for this project.")

manager = TodoManager()

while True:
    print("\n1. Add Task")
    print("2. Show All Tasks")
    print("3. Filter by Project")
    print("4. Exit")

    choice = input("Enter choice: ")

    if choice == "1":
        title = input("Enter task title: ")
        deadline = input("Enter deadline (YYYY-MM-DD): ")
        priority = input("Enter priority (High/Medium/Low): ")
        project = input("Enter project name: ")

        manager.add_task(title, deadline, priority, project)

    elif choice == "2":
        manager.show_all_tasks()

    elif choice == "3":
        project_name = input("Enter project name to filter: ")
        manager.filter_by_project(project_name)

    elif choice == "4":
        break

    else:
        print("Invalid choice")
